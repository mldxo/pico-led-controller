217const char* cgi_led_brightness_handler(int iIndex, int iNumParams, char* pcParam[], char* pcValue[])
218{
219 printf("Handling brightness request\n");
220 for (int i = 0; i < iNumParams; ++i)
221 {
222 int brightness = atoi(pcParam[i]);
223 if (brightness < 0 || brightness > 100)
224 printf("Invalid brightness value\n");
225 else
226 {
227 printf("Setting brightness to %d\n", brightness);
228 if (brightness == 0)
229 brightness = 0;
230 else if (brightness == 100)
231 brightness = 255;
232 else
233 {
234 brightness = (int)(brightness * 255.0 / 100.0) + 1;
235 if (brightness > 255)
236 brightness = 255;
237 else if (brightness < 0)
238 brightness = 0;
239 }
240 light_state.brightness = brightness;
241 }
242 if (light_state.light_mode == MODE_STATIC)
243 set_light_color(light_state.color);
244 }
245 return "/index.html";
246}
